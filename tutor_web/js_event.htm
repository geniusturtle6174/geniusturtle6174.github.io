前面的篇章中，我們曾提到過「onClick」和「onMouseover」等事件。事實上，還有許多常見的事件，可以增加網頁的互動性，舉例如下(如果你發現 onBeforeUnload 沒有作用，這是因為大部分的瀏覽器已經不允許該事件有自訂訊息。若要嘗試該事件的效果，可以使用 IE 瀏覽器)：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff" onLoad="alert('Hi~!')" onBeforeUnload="return('bye~!')">

<form>
	<input type="text" name="a" onChange="this.form.c.value=this.form.a.value"><br/>
	<input type="text" name="b" onKeyDown="this.form.c.value=this.form.b.value"><br/>
	<input type="text" name="c"><br/>
	<input type="text" name="d" onFocus="this.value='Hi~'" onBlur="this.value='Bye~'"><br/>
</form>

</body></html>

將事件內容寫在標籤裡面的寫法，叫做"事件屬性"。雖然標準上來說必須是小寫(例如： onclick)，但大多數的瀏覽器在解讀時，會允許大寫(onClick)。不過若是寫在 JavaScript 當中時，就必須按照語言規定，全部是小寫，並且必須以函式的形式指派事件內容：

<script>
document.getElementById('a').onclick = function(){
	console.log('Hi~~');
}
</script>

或者，用「addEventListener」來指派事件：

<script>
document.getElementById('a').addEventListener('click', function(){
	console.log('Hi~~');
}
</script>

除此之外，還存在其他的注意事項。例如你會發現這個範例不能動，這是因為事件在 HTML 文件結構還沒載入完成時就被指定的緣故：

﻿<html>
<head>
<script>
document.getElementById('a').onclick = function(){
	console.log('Hi~~');
}
</script>
</head>
<body bgcolor="#ccccff">

<form>
	<input type="button" id="a" value="按我">
</form>

</body></html>

若要解決這個問題，我們可以把 JavaScript 程式片段移到相關的物件後面，或者在文件結構甚至完整內容載入完成時才來指定事件：

﻿<html>
<head>
<script>
// load: 文件完整內容載入完成
window.onload = function(){
	document.getElementById('a').onclick = function(){
		console.log('This is a~');
	}
}

// DOMContentLoaded: 文件結構載入完成
document.addEventListener("DOMContentLoaded", function(){
	document.getElementById('b').onclick = function(){
		console.log('This is b~');
	}
});
</script>
</head>
<body bgcolor="#ccccff">

<form>
	<input type="button" id="a" value="按我(a)">
	<input type="button" id="b" value="按我(b)">
	<input type="button" id="c" value="按我(c)">
</form>

<script>
document.getElementById('c').onclick = function(){
	console.log('This is c~');
}
</script>

</body></html>

而在 JavaScript 當中，假若你指定了某個物件的 click 事件，則除了使用者真正點擊時會觸發事件以外，也可以透過呼叫該物件的 click 方法來觸發事件。若要測試，可在操作上述範例時打開 console，輸入「document.getElementById('a').click()」。

關於鍵盤中，我們可以利用以下事件：

  • onkeydown: 按下按鍵
  • onkeypress: 壓住按鍵
  • onkeyup: 放開按鍵

而若要知道是哪個鍵被按下，則可使用「event.keyCode」。範例如下：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<script>
	function myTest(){
		alert(event.keyCode);
		return false; // 如果要將某按鍵的功能取消，則可加上 return false
	}
	document.onkeyup = myTest;
</script>

</body></html>

偵測特殊按鍵以及複合鍵：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<script>
	function myTest(){
		if(event.shiftKey && event.keyCode!=16){
			alert("Shift + " + String.fromCharCode(event.keyCode));
		}
		if(event.ctrlKey && event.keyCode!=17){
			alert("Ctrl + " + String.fromCharCode(event.keyCode));
		}
		if(event.altKey && event.keyCode!=18){
			alert("Alt + " + String.fromCharCode(event.keyCode));
		}
	}
	document.onkeydown = myTest;
</script>

</body></html>

利用鍵盤來控制物件：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<script>
	function myTest(){
		sq = document.getElementById("mySquare");
		if(event.keyCode == 37){
			sq.style.left = parseInt(sq.style.left) - 50 + "px";
		}
		if(event.keyCode == 38){
			sq.style.top = parseInt(sq.style.top) - 50 + "px";
		}
		if(event.keyCode == 39){
			sq.style.left = parseInt(sq.style.left) + 50 + "px";
		}
		if(event.keyCode == 40){
			sq.style.top = parseInt(sq.style.top) + 50 + "px";
		}
	}
	document.onkeydown = myTest;
</script>

<div id="mySquare" style="position:absolute;top:20;left:20;width:50px;height:50px;background:red"></div>

</body></html>

跟著滑鼠游標動的方塊：

用來改變 style，則可以產生有趣的效果：

﻿<html>
<head>

		}
		#slideShowImg{
			position: absolute; top: 100px; left: 350px;
			z-index: 2;

	function showSlide(path){

		document.getElementById("slideShow").style.display = "block";
		document.getElementById("slideShowImg").style.display = "block";
		window.history.pushState(
			{"what_is_this": "隨便放一些你自己認得出 state 的東西"},

		);
	}
	
	function myHide(){
		document.getElementById("slideShow").style.display = "none";
		document.getElementById("slideShowImg").style.display = "none";
		window.history.back();
	}

	window.onpopstate = function(){myHide();}
</script>

如果你在一連串的物件上都綁上了相同名字的事件，則當觸發內層的物件事件時，會發現外層的事件也被觸發：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<ul id="list">
	<li id="item">Hello <a id="link" href="#">World</a>!</li>
</ul>

<script>

document.getElementById("list").onclick = function(){
	console.log("List clicked");
}

document.getElementById("item").onclick = function(){
	console.log("Item clicked");
}

document.getElementById("link").onclick = function(){
	console.log("Link clicked");
}

</script>

</body></html>

這是因為 JavaScript 的事件會在 DOM (Document Object Model)上傳遞，其順序是由上到下，再由下到上，前者稱為捕獲而後者稱為冒泡，觸發事件本身的元素則稱為目標。我們用 addEventListener 的第三個參數來展示，該參數可以指定是捕獲還是冒泡階段時執行，true 代表在捕獲階段執行，false 或不傳代表在冒泡階段執行。另外，由於目標階段不會區分捕獲還是冒泡，所以若你把加入事件的順序調換，則會發現 console 輸出的順序也會跟著調換：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<ul id="list">
	<li id="item">Hello <a id="link" href="#">World</a>!</li>
</ul>

<script>

document.getElementById("list").addEventListener(
	'click', function(){
		console.log("List clicked (capture)");
	}, true
);

document.getElementById("item").addEventListener(
	'click', function(){
		console.log("Item clicked (capture)");
	}, true
);

document.getElementById("link").addEventListener(
	'click', function(){
		console.log("Link clicked (capture)");
	}, true
);

document.getElementById("list").addEventListener(
	'click', function(){
		console.log("List clicked (bubble)");
	}, false
);

document.getElementById("item").addEventListener(
	'click', function(){
		console.log("Item clicked (bubble)");
	}
);

document.getElementById("link").addEventListener(
	'click', function(){
		console.log("Link clicked (bubble)");
	}
);

</script>

</body></html>

如果想要停止事件的傳遞，可以使用「stopPropagation」或者「stopImmediatePropagation」。兩者都可以阻止事件繼續傳遞，但使用前者時，同一層級的其他 listener 還是會被執行，而後者則否。以下範例示範「stopPropagation」的效果：

﻿<html>
<head>
</head>
<body bgcolor="#ccccff">

<ul id="list">
	<li id="item">Hello <a id="link" href="#">World</a>!</li>
</ul>

<script>

document.getElementById("list").addEventListener(
	'click', function(){
		console.log("List clicked (capture)");
	}, true
);

document.getElementById("item").addEventListener(
	'click', function(e){
		console.log("Item clicked (capture)");
		e.stopPropagation();
	}, true
);

document.getElementById("link").addEventListener(
	'click', function(){
		console.log("Link clicked (capture)");
	}, true
);

document.getElementById("list").addEventListener(
	'click', function(){
		console.log("List clicked (bubble)");
	}, false
);

document.getElementById("item").addEventListener(
	'click', function(){
		console.log("Item clicked (bubble)");
	}
);

document.getElementById("link").addEventListener(
	'click', function(){
		console.log("Link clicked (bubble)");
	}
);

</script>

</body></html>
