C語言當中，除了「if...else」以外，也可以用「?:」來進行條件判斷： 

#include <stdio.h>

int main()
{
	int a=3, b=4, c;

	c = (a>b)?(a-b):(b-a);
	printf("c: %d\n", c);

	return 0;
}

「#define」除了定義常數以外，也可以定義一些簡單的函式： 

#include <stdio.h>

#define MAX(a,b) ( ( (a) > (b) ) ? (a) : (b) )

int main()
{
	int a=3, b=4 , c;

	c = MAX(a,b);
	printf("c: %d\n", c);
	
	return 0;
}

變數宣告可以放在函式開頭以外的地方： 

#include <stdio.h>

int main()
{
	int a=3;

	if(a>=2){
		int a=4;
		printf("a: %d\n", a);
	}
	printf("a: %d\n", a);

	return 0;
}
如果放在迴圈裡面，則每次都會被重新初始化： 
#include <stdio.h>

int main()
{
	int i;

	for(i=0;i<5;i++){
		int a=4;
		printf("a: %d\n", a);
		a++;
	}

	return 0;
}

「scanf」會回傳"讀到了幾樣東西"： 

#include <stdio.h>

int main()
{
	int a, b, c;

	c = scanf("%d %d", &a, &b);
	printf("c: %d\n", c);

	return 0;
}

「stdin」和「stdout」其實也是檔案指標： 

#include <stdio.h>

int main()
{
	int a;

	fscanf(stdin, "%d", &a);
	fprintf(stdout, "a: %d\n", a);
	
	return 0;
}
